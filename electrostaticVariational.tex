  problem PoissonVarf(phi,w,solver=CG) =
      int2d(Allh)(
   	   EPS[0][0] * dx(phi)*dx(w)
   	 + EPS[0][1] * dx(phi)*dy(w)
   	 + EPS[1][0] * dy(phi)*dx(w)
   	 + EPS[1][1] * dy(phi)*dy(w) )
   	 - int2d(Allh)( source*w )
   	 + on(Lboxup,phi = phiCup)
   	 + on(Lboxdown,phi = phiCdown)
    ;
  PoissonVarf;
